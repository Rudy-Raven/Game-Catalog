create table User (user_id int NOT NULL, firstname varchar(100) NOT NULL, lastname varchar(100) NOT NULL, password varchar(20) NOT NULL, email varchar(30) NOT NULL, dob date NOT NULL, list_id int, PRIMARY KEY(user_id), foreign key (list_id) references game_list(list_id));

create table friends (user_id1 int, user_id2 int, PRIMARY KEY(user_id1,user_id2), foreign key(user_id1) references User(user_id), foreign key(user_id2) references User(user_id));
